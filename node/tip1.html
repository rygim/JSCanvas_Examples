Your Game loop:
Simple animation loop (Don't do this) 
while(true){
    update();
    draw();
};
Actual animation loop (OK) 
var loop = function(){
    update();
    draw();
};
setInterval(loop, 10);
Better animation loop 
var lastDrawTime = new Date().getTime();
var drawLoop = function(){
    var startDrawTime = new Date().getTime(); 
    interpolate(startDrawTime - lastDrawTime);
    lastDrawTime = startDrawTime;
    draw();
    setTimeout(drawLoop, 5);
};
drawLoop();
setInterval(update, 40);
Future animation loop 
var lastDrawTime = new Date().getTime();
var drawLoop = function(){
    var startDrawTime = new Date().getTime(); 
    interpolate(startDrawTime - lastDrawTime);
    lastDrawTime = startDrawTime;
    draw();
    requestAnimationFrame(drawLoop);
};
setInterval(update, 40);
